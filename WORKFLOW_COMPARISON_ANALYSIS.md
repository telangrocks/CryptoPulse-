```javascript
function recordTrading(pair, strategy, action, amount, success, profit) {
  const metric = {
    pair,
    strategy,
    action,
    amount,
    success,
    profit,
    timestamp: new Date()
  };
  
  metrics.trading.push(metric);
  
  // Keep only last 1000 trading records
  if (metrics.trading.length > 1000) {
    metrics.trading = metrics.trading.slice(-1000);
  }
}
```
